**Код на языке С++**  
```c++
#include <iostream>
#include <cmath>

int main()
{
    double N = 7; // angles
    double n = 2; // приёмы
    double m_beta = 5; // seconds

    double f = 2 * sqrt(N / n * pow(m_beta,2));
    std::cout << f << std::endl;   
}
```
